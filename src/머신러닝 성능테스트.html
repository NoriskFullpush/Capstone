from sklearn.metrics import recall_score

def eval_xgb(model, X_valid, y_valid):
    y_pred = model.predict(X_valid)
    return recall_score(y_valid, y_pred, pos_label=0)

from xgboost import XGBClassifier

learning_rates = [0.1, 0.05, 0.01] 

results = []

for lr in learning_rates:
    xgb = XGBClassifier(
        n_estimators=5000,    # 충분히 큰 수로 설정
        learning_rate=lr,     
        max_depth=7,
        min_child_weight=3,
        subsample=0.8,
        colsample_bytree=0.8,
        random_state=42,
        early_stopping_rounds=50,
        n_jobs=-1,
    )
    
    xgb.fit(
        X_tr, y_tr,
        eval_set=[(X_valid, y_valid)],
        verbose=False
    )
 
    score = eval_xgb(xgb, X_valid, y_valid)
    
    best_iter = xgb.best_iteration
